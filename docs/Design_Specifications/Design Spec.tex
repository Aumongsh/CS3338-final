\section{Website Breakdown}

\subsection{Homepage}
\begin{itemize}
    \item Input fields for two Youtube channel URLs.
    \item Select number of videos and filter method.
    \item 'Compare' button to start backend process.
\end{itemize}

\subsection{Results Page}
\begin{itemize}
    \item Shows shared commenter count.
    \item Stats & Visualization.
\end{itemize}


\section{Functionalities}

\subsection{Frontend}
\begin{itemize}
    \item Styling, UI
    \item Results display
\end{itemize}

\subsection{Backend}
\begin{itemize}
    \item Validate channel input
    \item Grab videos and comments via YouTube API
    \item Save data into database
    \item Calculate number of shared commenters
    \item Return results to frontend
\end{itemize}
